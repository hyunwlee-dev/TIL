# <mark>BOJ 2164 카드2</mark>

```
class HW_queue {
    constructor() {
        this.arr = [];
    }
    offer(item){
        this.arr.push(item);
    }
    poll(){
        return (this.arr.shift());
    }
    peek(){
        return (this.arr[0]);
    }
    isEmpty(){
        return (this.arr.length === 0);
    }
}

class Node {
    constructor(value) {
        this.value = value;
        this.next = null;
    }
}

class LinkedList{
    constructor() {
        this.head = null;
        this.tail = null;
        this.size = 0;
    }
    offer(item) {
        const newNode = new Node(item);
        if (!this.head)
        {
            this.head = newNode;
            this.tail = newNode;
        }
        else
        {
            this.tail.next = newNode;
            this.tail = newNode;
        }
        ++this.size;
    }
    poll() {
        if (this.isEmpty())
            return;
        const item = this.head.value;
        this.head = this.head.next;
        if (!this.head)
            this.tail = null;
        --this.size;
        return (item);
    }
    peek(){
        return (this.head.value);
    }
    isEmpty() {
        return (this.head === null);
    }
}

function solution(input) {
    const n = +input[0];
    const queue = new LinkedList();
    for (let num = 1; num <= n; ++num)
        queue.offer(num);
    while (queue.size > 1)
    {
        queue.poll();
        queue.offer(queue.poll());
    }
    console.log(queue.peek());
}
```

n이 최대 500,000이기 때문에 O(n<sup>2</sup>)의 시간복잡도가 일어난다면 시간초과가 나게된다. js의 array의 poll()기능 shift를 쓰면 O(n)이나 걸린다. 따라서 따로 linkedList를 구현하여 poll()의 기능 O(1)이 나도록 하게 작성하면 해결된다.  

O(n<sup>2</sup>) => O(n)

function mergeSort(arr, start, end) {
    if (start >= end)
        return;
    let mid = Math.floor((start + end) / 2);
    mergeSort(arr, start, mid);
    mergeSort(arr, mid + 1, end);
    merge(arr, start, mid, end);
}

function merge(arr, start, mid, end) {
    let ret = Array.from({length: (mid - start + 1) + (end - mid + 1 + 1)}, (item) => 0);
    let retIdx = 0;
    let ltIdx = start;
    let rtIdx = mid + 1;
    while (ltIdx <= mid && rtIdx <= end)
    {
        if (arr[ltIdx] <= arr[rtIdx])
            ret[retIdx++] = arr[ltIdx++];
        else
            ret[retIdx++] = arr[rtIdx++];
    }
    while (ltIdx <= mid)
        ret[retIdx++] = arr[ltIdx++];
    while (rtIdx <= end)
        ret[retIdx++] = arr[rtIdx++];
    for (let i = start; i <= end; ++i)
        arr[i] = ret[i - start];
}

let max = 0;

function solution(input) {
    const[n, m] = input[0].split(' ').map((item) => +item);
    const arr = input[1].split(' ').map((item) => +item);
    const check = Array.from({length: n}, item => false);
    const temp = Array.from({length: 3}, item => 0);
    recurse(0, arr, temp, check, m);
    console.log(max);
}

function recurse(depth, arr, temp, check, target) {
    if (depth === 3)
    {
        const val = Math.max(max, temp.reduce((cal, cur) => (cal + arr[cur]), 0));
        if (val <= target)
            max = val;
        return;
    }
    for (let i = 0; i <= arr.length - 1; ++i)
    {
        if (check[i])
            continue;
        check[i] = true;
        temp[depth] = i;
        recurse(depth + 1, arr, temp, check, target);
        check[i] = false;
    }
}
