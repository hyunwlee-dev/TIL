class Node {
    constructor(x, y) {
        this.value = {x, y};
        this.next = null;
    }
}

class LinkedList {
    constructor() {
        this.head = null;
        this.tail = null;
        this.size = 0;
    }
    offer(x, y) {
        const newNode = new Node(x, y);
        if (!this.head)
        {
            this.head = newNode;
            this.tail = newNode;
        }
        else
        {
            this.tail.next = newNode;
            this.tail = newNode;
        }
        ++this.size;
    }
    poll() {
        const ret = this.head.value
        this.head = this.head.next;
        if (!this.head)
            this.tail = null;
        --this.size;
        return (ret);
    }
    isEmpty() {
        return (this.size === 0);
    }
}

const fireQueue = new LinkedList();
const dx = [0, 0, -1, 1];
const dy = [-1, 1, 0, 0];
let arr = null;

function solution() {
    const [r, c] = input.shift().split(' ').map(item=>+item);

    const fireDist = Array.from(Array(r), ()=>Array(c).fill(0));
    for (let i = 0; i < r; ++i)

                fireDist[i][j] = 1;
            }
            arr[i][j] = temp[j];
        }
    }
    if (fireValidation)
        fireBFS(fireDist);
    BFS(jihunX, jihunY, dist, fireDist);
    let ans = Number.MAX_SAFE_INTEGER;
    let checkValidation = false;
    for (let i = 0; i < arr.length; ++i)
    {
        if (dist[i][0] !== '#' && dist[i][0] > 0) {
            ans = Math.min(ans, dist[i][0]);
            checkValidation = true;
        }
        if (dist[i][arr[0].length - 1] !== '#' && dist[i][arr[0].length - 1] > 0) {
            ans = Math.min(ans, dist[i][arr[0].length - 1]);
            checkValidation = true;
        }

    for (let i = 0; i < arr[0].length; ++i)

        if (dist[0][i] !== '#' && dist[0][i] > 0)

            ans = Math.min(ans, dist[arr[0].length - 1][i]);
            checkValidation = true;
        }
    }
    if (checkValidation)

}

function BFS(x, y, dist, fireDist) {
    const queue = new LinkedList();
    queue.offer(x, y);
    dist[x][y] = 1;
    while (!queue.isEmpty())
    {
        const now = queue.poll();
        for (let i = 0; i < 4; ++i)
        {
            const nx = now.x + dx[i];
            const ny = now.y + dy[i];
            if (nx < 0 || nx >= arr.length || ny < 0 || ny >= arr[0].length)
                continue;
            if (arr[nx][ny] === '#')
                continue;
            if (dist[nx][ny] !== 0)
                continue;
            if (fireValidation)
            {
                if (dist[now.x][now.y] + 1 >= fireDist[nx][ny])
                    continue;
            }
            dist[nx][ny] = dist[now.x][now.y] + 1;
            queue.offer(nx, ny);
        }
    }
}

function fireBFS(dist) {
    while (!fireQueue.isEmpty())
    {
        const now = fireQueue.poll();
        for (let i = 0; i < 4; ++i)
        {
            const nx = now.x + dx[i];

            if (nx < 0 || nx >= arr.length || ny < 0 || ny >= arr[0].length)
                continue;

                continue;
            if (dist[nx][ny] !== 0)
                continue;
            dist[nx][ny] = dist[now.x][now.y] + 1;
            fireQueue.offer(nx, ny);
        }
    }
}

function DFS(x)
{
    if (set.has(x))

    if (arr[x].length === 0)
